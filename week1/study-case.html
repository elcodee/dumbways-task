Study Case

Type Age
Bayi 0 - 1
Anak - Anak 2 - 10
Remaja 11 - 19
Dewasa 20 - 60
Lanjut Usia E

Input : 3
Output : Umur 3 tahun adalah Anak-anak

Code

<input type="number" oninput="age(event.target.value)" class="form-control" placeholder="masukan umur">

const age = (num) => {
if (num <= 1) {
    document.getElementById("display").innerHTML = "Umur " + num + " Adalah Bayi";
}
if (num >= 2 && num <= 10) {
    document.getElementById("display").innerHTML = "Umur " + num + " Adalah Anak-anak";
}
if (num >= 11 && num <= 19) {
    document.getElementById("display").innerHTML = "Umur " + num + " Adalah Remaja";
}
if (num >= 20 && num <= 60) {
    document.getElementById("display").innerHTML = "Umur " + num + " Adalah Dewasa";
  }
